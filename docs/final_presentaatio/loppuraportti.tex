\begin{frame}{Tilanne tällä hetkellä}
\begin{itemize}
\item RC-autoon suunniteltu uusi voimansiirto
\item Nykyisellä voimansiirrolla ajaminen / testaaminen ei onnistu
\item Tästä johtuen mitään testausta ei ole liikkuvalla RC-autolla tehty
\item Projekti lähtenyt vähän hitaasti liikkeelle
\end{itemize}
\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{ABS- ja ESC -järjestelmät}
\begin{itemize}
\item Mittauksia nykyisestä ohjelmistosta tehty
\item ABS-järjestelmän toimimattomuuden syynä hidas mekaniikka (jarruservot)
\item Servon päivitystahti 20 ms
\item Digiservoja/OpenServo? Jotain muuta?
\item Vaijerien venyminen, palautusjouset, ...
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Käyttöliittymä}
\begin{itemize}

\item Protokollamäärittelyt joustavampia
\item Jatkokehitys helpompaa
\end{itemize}
\end{frame}

\end{itemize}
